1488        /**
1489         * This method is used to perform argument checking on the list of
1490         * axis indices passed to mapDatasetToDomainAxes() and
1491         * mapDatasetToRangeAxes().
1492         *
1493         * @param indices  the list of indices (<code>null</code> permitted).
1494         */
1495        private void checkAxisIndices(List indices) {
1496            // axisIndices can be:
1497            // 1.  null;
1498            // 2.  non-empty, containing only Integer objects that are unique.
1499            if (indices == null) {
1500                return;  // OK
1501            }
1502            int count = indices.size();
1503            if (count == 0) {
1504                throw new IllegalArgumentException("Empty list not permitted.");
1505            }
1506            HashSet set = new HashSet();
1507            for (int i = 0; i < count; i++) {
1508                Object item = indices.get(i);
1509                if (!(item instanceof Integer)) {
1510                    throw new IllegalArgumentException(
1511                            "Indices must be Integer instances.");
1512                }
1513                if (set.contains(item)) {
1514                    throw new IllegalArgumentException("Indices must be unique.");
1515                }
1516                set.add(item);
1517            }
1518        }

4377         *
4378         * @param axis  the axis.
4379         *
4380         * @return The range.
4381         */
4382        public Range getDataRange(ValueAxis axis) {
4383    
4384            Range result = null;
4385            List mappedDatasets = new ArrayList();
4386            boolean isDomainAxis = true;
4387    
4388            // is it a domain axis?
4389            int domainIndex = getDomainAxisIndex(axis);
4390            if (domainIndex >= 0) {
4391                isDomainAxis = true;
4392                mappedDatasets.addAll(getDatasetsMappedToDomainAxis(
4393                        new Integer(domainIndex)));
4394            }
4395    
4396            // or is it a range axis?
4397            int rangeIndex = getRangeAxisIndex(axis);
4398            if (rangeIndex >= 0) {
4399                isDomainAxis = false;
4400                mappedDatasets.addAll(getDatasetsMappedToRangeAxis(
4401                        new Integer(rangeIndex)));
4402            }
4403    
4404            // iterate through the datasets that map to the axis and get the union
4405            // of the ranges.
4406            Iterator iterator = mappedDatasets.iterator();
4407            while (iterator.hasNext()) {
4408                XYDataset d = (XYDataset) iterator.next();
4409                if (d != null) {
4410                    XYItemRenderer r = getRendererForDataset(d);
4411                    if (isDomainAxis) {
4412                        if (r != null) {
4413                            result = Range.combine(result, r.findDomainBounds(d));
4414                        }
4415                        else {
4416                            result = Range.combine(result,
4417                                    DatasetUtilities.findDomainBounds(d));
4418                        }
4419                    }
4420                    else {
4421                        if (r != null) {
4422                            result = Range.combine(result, r.findRangeBounds(d));
4423                        }
4424                        else {
4425                            result = Range.combine(result,
4426                                    DatasetUtilities.findRangeBounds(d));
4427                        }
4428                    }
4429                }
4430            }
4431            return result;

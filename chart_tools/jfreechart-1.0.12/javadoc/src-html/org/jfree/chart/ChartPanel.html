544        /**
545         * Constructs a JFreeChart panel.
546         *
547         * @param chart  the chart.
548         * @param width  the preferred width of the panel.
549         * @param height  the preferred height of the panel.
550         * @param minimumDrawWidth  the minimum drawing width.
551         * @param minimumDrawHeight  the minimum drawing height.
552         * @param maximumDrawWidth  the maximum drawing width.
553         * @param maximumDrawHeight  the maximum drawing height.
554         * @param useBuffer  a flag that indicates whether to use the off-screen
555         *                   buffer to improve performance (at the expense of
556         *                   memory).
557         * @param properties  a flag indicating whether or not the chart property
558         *                    editor should be available via the popup menu.
559         * @param save  a flag indicating whether or not save options should be
560         *              available via the popup menu.
561         * @param print  a flag indicating whether or not the print option
562         *               should be available via the popup menu.
563         * @param zoom  a flag indicating whether or not zoom options should be
564         *              added to the popup menu.
565         * @param tooltips  a flag indicating whether or not tooltips should be
566         *                  enabled for the chart.
567         */
568        public ChartPanel(JFreeChart chart,
569                          int width,
570                          int height,
571                          int minimumDrawWidth,
572                          int minimumDrawHeight,
573                          int maximumDrawWidth,
574                          int maximumDrawHeight,
575                          boolean useBuffer,
576                          boolean properties,
577                          boolean save,
578                          boolean print,
579                          boolean zoom,
580                          boolean tooltips) {
581    
582            setChart(chart);
583            this.chartMouseListeners = new EventListenerList();
584            this.info = new ChartRenderingInfo();
585            setPreferredSize(new Dimension(width, height));
586            this.useBuffer = useBuffer;
587            this.refreshBuffer = false;
588            this.minimumDrawWidth = minimumDrawWidth;
589            this.minimumDrawHeight = minimumDrawHeight;
590            this.maximumDrawWidth = maximumDrawWidth;
591            this.maximumDrawHeight = maximumDrawHeight;
592            this.zoomTriggerDistance = DEFAULT_ZOOM_TRIGGER_DISTANCE;
593    
594            // set up popup menu...
595            this.popup = null;
596            if (properties || save || print || zoom) {
597                this.popup = createPopupMenu(properties, save, print, zoom);
598            }
599    
600            enableEvents(AWTEvent.MOUSE_EVENT_MASK);
601            enableEvents(AWTEvent.MOUSE_MOTION_EVENT_MASK);
602            setDisplayToolTips(tooltips);
603            addMouseListener(this);
604            addMouseMotionListener(this);
605    
606            this.defaultDirectoryForSaveAs = null;
607            this.enforceFileExtensions = true;
608    
609            // initialize ChartPanel-specific tool tip delays with
610            // values the from ToolTipManager.sharedInstance()
611            ToolTipManager ttm = ToolTipManager.sharedInstance();
612            this.ownToolTipInitialDelay = ttm.getInitialDelay();
613            this.ownToolTipDismissDelay = ttm.getDismissDelay();
614            this.ownToolTipReshowDelay = ttm.getReshowDelay();
615    
616            this.zoomAroundAnchor = false;
617        }

628        /**
629         * Sets the chart that is displayed in the panel.
630         *
631         * @param chart  the chart (<code>null</code> permitted).
632         */
633        public void setChart(JFreeChart chart) {
634    
635            // stop listening for changes to the existing chart
636            if (this.chart != null) {
637                this.chart.removeChangeListener(this);
638                this.chart.removeProgressListener(this);
639            }
640    
641            // add the new chart

1245        /**
1246         * Paints the component by drawing the chart to fill the entire component,
1247         * but allowing for the insets (which will be non-zero if a border has been
1248         * set for this component).  To increase performance (at the expense of
1249         * memory), an off-screen buffer image can be used.
1250         *
1251         * @param g  the graphics device for drawing on.
1252         */
1253        public void paintComponent(Graphics g) {
1254            super.paintComponent(g);
1255            if (this.chart == null) {
1256                return;
1257            }
1258            Graphics2D g2 = (Graphics2D) g.create();
1259    
1260            // first determine the size of the chart rendering area...
1261            Dimension size = getSize();
1262            Insets insets = getInsets();
1263            Rectangle2D available = new Rectangle2D.Double(insets.left, insets.top,
1264                    size.getWidth() - insets.left - insets.right,
1265                    size.getHeight() - insets.top - insets.bottom);
1266    
1267            // work out if scaling is required...
1268            boolean scale = false;
1269            double drawWidth = available.getWidth();
1270            double drawHeight = available.getHeight();
1271            this.scaleX = 1.0;
1272            this.scaleY = 1.0;
1273    
1274            if (drawWidth < this.minimumDrawWidth) {
1275                this.scaleX = drawWidth / this.minimumDrawWidth;
1276                drawWidth = this.minimumDrawWidth;
1277                scale = true;
1278            }
1279            else if (drawWidth > this.maximumDrawWidth) {
1280                this.scaleX = drawWidth / this.maximumDrawWidth;
1281                drawWidth = this.maximumDrawWidth;
1282                scale = true;
1283            }
1284    
1285            if (drawHeight < this.minimumDrawHeight) {
1286                this.scaleY = drawHeight / this.minimumDrawHeight;
1287                drawHeight = this.minimumDrawHeight;
1288                scale = true;
1289            }
1290            else if (drawHeight > this.maximumDrawHeight) {
1291                this.scaleY = drawHeight / this.maximumDrawHeight;
1292                drawHeight = this.maximumDrawHeight;
1293                scale = true;
1294            }
1295    
1296            Rectangle2D chartArea = new Rectangle2D.Double(0.0, 0.0, drawWidth,
1297                    drawHeight);
1298    
1299            // are we using the chart buffer?
1300            if (this.useBuffer) {
1301    
1302                // if buffer is being refreshed, it needs clearing unless it is
1303                // new - use the following flag to track this...
1304                boolean clearBuffer = true;
1305    
1306                // do we need to resize the buffer?
1307                if ((this.chartBuffer == null)
1308                        || (this.chartBufferWidth != available.getWidth())
1309                        || (this.chartBufferHeight != available.getHeight())) {
1310                    this.chartBufferWidth = (int) available.getWidth();
1311                    this.chartBufferHeight = (int) available.getHeight();
1312                    GraphicsConfiguration gc = g2.getDeviceConfiguration();
1313                    this.chartBuffer = gc.createCompatibleImage(
1314                            this.chartBufferWidth, this.chartBufferHeight,
1315                            Transparency.TRANSLUCENT);
1316                    this.refreshBuffer = true;
1317                    clearBuffer = false;  // buffer is new, no clearing required
1318                }
1319    
1320                // do we need to redraw the buffer?
1321                if (this.refreshBuffer) {
1322    
1323                    this.refreshBuffer = false; // clear the flag
1324    
1325                    Rectangle2D bufferArea = new Rectangle2D.Double(
1326                            0, 0, this.chartBufferWidth, this.chartBufferHeight);
1327    
1328                    Graphics2D bufferG2 = (Graphics2D)
1329                            this.chartBuffer.getGraphics();
1330                    if (clearBuffer) {
1331                        bufferG2.clearRect(0, 0, this.chartBufferWidth,
1332                                this.chartBufferHeight);
1333                    }
1334                    if (scale) {
1335                        AffineTransform saved = bufferG2.getTransform();
1336                        AffineTransform st = AffineTransform.getScaleInstance(
1337                                this.scaleX, this.scaleY);
1338                        bufferG2.transform(st);
1339                        this.chart.draw(bufferG2, chartArea, this.anchor,
1340                                this.info);
1341                        bufferG2.setTransform(saved);
1342                    }
1343                    else {
1344                        this.chart.draw(bufferG2, bufferArea, this.anchor,
1345                                this.info);
1346                    }
1347    
1348                }
1349    
1350                // zap the buffer onto the panel...
1351                g2.drawImage(this.chartBuffer, insets.left, insets.top, this);
1352    
1353            }
1354    
1355            // or redrawing the chart every time...
1356            else {
1357    
1358                AffineTransform saved = g2.getTransform();
1359                g2.translate(insets.left, insets.top);
1360                if (scale) {
1361                    AffineTransform st = AffineTransform.getScaleInstance(
1362                            this.scaleX, this.scaleY);
1363                    g2.transform(st);
1364                }
1365                this.chart.draw(g2, chartArea, this.anchor, this.info);
1366                g2.setTransform(saved);
1367    
1368            }
1369    
1370            // Redraw the zoom rectangle (if present)
1371            drawZoomRectangle(g2);
1372    
1373            g2.dispose();
1374    
1375            this.anchor = null;
1376            this.verticalTraceLine = null;

1552        /**
1553         * Handles a 'mouse dragged' event.
1554         *
1555         * @param e  the mouse event.
1556         */
1557        public void mouseDragged(MouseEvent e) {
1558    
1559            // if the popup menu has already been triggered, then ignore dragging...
1560            if (this.popup != null && this.popup.isShowing()) {
1561                return;
1562            }
1563            // if no initial zoom point was set, ignore dragging...
1564            if (this.zoomPoint == null) {
1565                return;
1566            }
1567            Graphics2D g2 = (Graphics2D) getGraphics();
1568    
1569            // Erase the previous zoom rectangle (if any)...
1570            drawZoomRectangle(g2);
1571    
1572            boolean hZoom = false;
1573            boolean vZoom = false;
1574            if (this.orientation == PlotOrientation.HORIZONTAL) {
1575                hZoom = this.rangeZoomable;
1576                vZoom = this.domainZoomable;
1577            }
1578            else {
1579                hZoom = this.domainZoomable;
1580                vZoom = this.rangeZoomable;
1581            }
1582            Rectangle2D scaledDataArea = getScreenDataArea(
1583                    (int) this.zoomPoint.getX(), (int) this.zoomPoint.getY());
1584            if (hZoom && vZoom) {
1585                // selected rectangle shouldn't extend outside the data area...
1586                double xmax = Math.min(e.getX(), scaledDataArea.getMaxX());
1587                double ymax = Math.min(e.getY(), scaledDataArea.getMaxY());
1588                this.zoomRectangle = new Rectangle2D.Double(
1589                        this.zoomPoint.getX(), this.zoomPoint.getY(),
1590                        xmax - this.zoomPoint.getX(), ymax - this.zoomPoint.getY());
1591            }
1592            else if (hZoom) {
1593                double xmax = Math.min(e.getX(), scaledDataArea.getMaxX());
1594                this.zoomRectangle = new Rectangle2D.Double(
1595                        this.zoomPoint.getX(), scaledDataArea.getMinY(),
1596                        xmax - this.zoomPoint.getX(), scaledDataArea.getHeight());
1597            }
1598            else if (vZoom) {
1599                double ymax = Math.min(e.getY(), scaledDataArea.getMaxY());
1600                this.zoomRectangle = new Rectangle2D.Double(
1601                        scaledDataArea.getMinX(), this.zoomPoint.getY(),
1602                        scaledDataArea.getWidth(), ymax - this.zoomPoint.getY());
1603            }
1604    
1605            // Draw the new zoom rectangle...
1606            drawZoomRectangle(g2);
1607    
1608            g2.dispose();
1609    
1610        }

1612        /**
1613         * Handles a 'mouse released' event.  On Windows, we need to check if this
1614         * is a popup trigger, but only if we haven't already been tracking a zoom
1615         * rectangle.
1616         *
1617         * @param e  information about the event.
1618         */
1619        public void mouseReleased(MouseEvent e) {
1620    
1621            if (this.zoomRectangle != null) {
1622                boolean hZoom = false;
1623                boolean vZoom = false;
1624                if (this.orientation == PlotOrientation.HORIZONTAL) {
1625                    hZoom = this.rangeZoomable;
1626                    vZoom = this.domainZoomable;
1627                }
1628                else {
1629                    hZoom = this.domainZoomable;
1630                    vZoom = this.rangeZoomable;
1631                }
1632    
1633                boolean zoomTrigger1 = hZoom && Math.abs(e.getX()
1634                    - this.zoomPoint.getX()) >= this.zoomTriggerDistance;
1635                boolean zoomTrigger2 = vZoom && Math.abs(e.getY()
1636                    - this.zoomPoint.getY()) >= this.zoomTriggerDistance;
1637                if (zoomTrigger1 || zoomTrigger2) {
1638                    if ((hZoom && (e.getX() < this.zoomPoint.getX()))
1639                        || (vZoom && (e.getY() < this.zoomPoint.getY()))) {
1640                        restoreAutoBounds();
1641                    }
1642                    else {
1643                        double x, y, w, h;
1644                        Rectangle2D screenDataArea = getScreenDataArea(
1645                                (int) this.zoomPoint.getX(),
1646                                (int) this.zoomPoint.getY());
1647                        double maxX = screenDataArea.getMaxX();
1648                        double maxY = screenDataArea.getMaxY();
1649                        // for mouseReleased event, (horizontalZoom || verticalZoom)
1650                        // will be true, so we can just test for either being false;
1651                        // otherwise both are true
1652                        if (!vZoom) {
1653                            x = this.zoomPoint.getX();
1654                            y = screenDataArea.getMinY();
1655                            w = Math.min(this.zoomRectangle.getWidth(),
1656                                    maxX - this.zoomPoint.getX());
1657                            h = screenDataArea.getHeight();
1658                        }
1659                        else if (!hZoom) {
1660                            x = screenDataArea.getMinX();
1661                            y = this.zoomPoint.getY();
1662                            w = screenDataArea.getWidth();
1663                            h = Math.min(this.zoomRectangle.getHeight(),
1664                                    maxY - this.zoomPoint.getY());
1665                        }
1666                        else {
1667                            x = this.zoomPoint.getX();
1668                            y = this.zoomPoint.getY();
1669                            w = Math.min(this.zoomRectangle.getWidth(),
1670                                    maxX - this.zoomPoint.getX());
1671                            h = Math.min(this.zoomRectangle.getHeight(),
1672                                    maxY - this.zoomPoint.getY());
1673                        }
1674                        Rectangle2D zoomArea = new Rectangle2D.Double(x, y, w, h);
1675                        zoom(zoomArea);
1676                    }
1677                    this.zoomPoint = null;
1678                    this.zoomRectangle = null;
1679                }
1680                else {
1681                    // Erase the zoom rectangle
1682                    Graphics2D g2 = (Graphics2D) getGraphics();
1683                    drawZoomRectangle(g2);
1684                    g2.dispose();
1685                    this.zoomPoint = null;
1686                    this.zoomRectangle = null;
1687                }
1688    
1689            }
1690    
1691            else if (e.isPopupTrigger()) {
1692                if (this.popup != null) {
1693                    displayPopupMenu(e.getX(), e.getY());
1694                }
1695            }
1696    
1697        }
